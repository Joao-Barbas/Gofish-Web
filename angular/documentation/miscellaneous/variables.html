Type : HttpInterceptorFn
Default value : (req, next) => { const authService: AuthService = inject(AuthService); const router = inject(Router); if (authService.isAuthenticated()) { req = req.clone({ headers: req.headers.set('Authorization', `Bearer ${authService.getEncodedToken()}`) }); } return next(req).pipe(catchError(err => { switch (err.status) { case 401: { // authService.removeToken(); // router.navigate([Path.SIGN_IN]); break; } case 403: { // router.navigate([Path.HOME]); break; }} return throwError(() => err); })); }
